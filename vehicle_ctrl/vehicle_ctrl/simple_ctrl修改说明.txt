def find_target_waypoint_improved(self):
    """改进的目标点选择算法 - 专门优化转弯"""
    if not self.waypoints_list or not self.odom:
        return self.target_waypoint_index
        
    vehicle_loc = self.odom.location
    
    # 计算当前路径曲率 - 这决定转弯时机
    current_curvature = self.calculate_path_curvature(self.target_waypoint_index)
    self.last_curvature = current_curvature
    
    # 检测是否进入弯道 - 转弯开始时机
    if current_curvature > 15 and not self.in_curve:  # ← 调整这个阈值来控制转弯检测灵敏度
        self.in_curve = True
        self.curve_start_index = self.target_waypoint_index
        self.get_logger().info(f"Entering curve at WP {self.target_waypoint_index}, curvature: {current_curvature:.1f}°")
    elif current_curvature < 8 and self.in_curve:  # ← 调整这个阈值来控制转弯结束时机
        self.in_curve = False
        self.get_logger().info(f"Exiting curve at WP {self.target_waypoint_index}")
    
    # 计算自适应前瞻距离 - 影响转弯提前量
    adaptive_lookahead = self.calculate_adaptive_lookahead(current_curvature)
    
    # 在弯道中，我们选择更近的点来确保跟踪精度
    if self.in_curve:
        # 弯道中：搜索范围较小，选择更近的点
        search_end = min(len(self.waypoints_list), start_index + 15)  # ← 调整这个值来控制弯道中的搜索范围
        min_dist = float('inf')
        
        for i in range(start_index, search_end):
            waypoint = self.waypoints_list[i]
            dist = self.calculate_distance(vehicle_loc, waypoint.location)
            
            # 在弯道中，我们选择距离最接近自适应前瞻距离的点
            dist_diff = abs(dist - adaptive_lookahead)
            if dist_diff < min_dist:
                min_dist = dist_diff
                target_index = i

def setup_PID_improved(self):
    """改进的PID参数设置，特别优化转向控制"""
    # 横向控制参数（转向控制）- 这些参数直接影响转弯角度
    args_lateral_dict = {
        'K_P': 0.4,   # ← 比例增益：值越大，转向响应越快，转弯越激进
        'K_D': 0.1,   # ← 微分增益：值越大，转向振荡越小，转弯越平滑
        'K_I': 0.002, # ← 积分增益：值越大，稳态误差修正越快
        'dt': self.control_period
    }

    PID = VehiclePIDController(
        self.vehicle,
        args_lateral=args_lateral_dict,
        args_longitudinal=args_long_dict,
        offset=0.5,             # ← 这是控制离路边距离的主要参数
        max_throttle=0.4,
        max_brake=0.5,
        max_steering=0.5  # ← 最大转向角度限制：值越大，允许的转弯角度越大
    )
    return PID

def smooth_steering_improved(self, new_steer, curvature):
    """改进的转向平滑，考虑路径曲率"""
    # 弯道时减少平滑，直道时增加平滑 - 这影响转弯的平滑度
    if curvature > 30:
        smoothing = self.steer_smoothing_factor * 0.7  # ← 弯道时减少平滑，转向更直接
    elif curvature > 15:
        smoothing = self.steer_smoothing_factor * 0.85
    else:
        smoothing = self.steer_smoothing_factor * 1.2  # ← 直道时增加平滑，转向更柔和
    
    smoothed_steer = (self.last_steer * (1 - smoothing) + 
                     new_steer * smoothing)
    self.last_steer = smoothed_steer
    return smoothed_steer

def calculate_turn_speed(self, target_index):
    """根据转弯程度计算合适的速度"""
    base_speed = 25  # km/h ← 调整这个基础速度
    
    # 计算路径曲率
    curvature = self.calculate_path_curvature(target_index)
    
    # 根据曲率调整速度 - 这些阈值影响不同弯道的速度
    if curvature > 50:  # ← 急弯阈值
        return base_speed * 0.4  # ← 急弯速度比例
    elif curvature > 30:  # ← 中等弯道阈值
        return base_speed * 0.6  # ← 中等弯道速度比例
    elif curvature > 15:  # ← 缓弯阈值
        return base_speed * 0.8  # ← 缓弯速度比例
    else:  # 直道
        return base_speed


4..前瞻距离影响转弯的提前量：



def calculate_adaptive_lookahead(self, curvature):
    """根据曲率自适应调整前瞻距离"""
    base_lookahead = 5.0  # ← 基础前瞻距离
    
    # 弯道时减少前瞻距离，直道时保持或增加
    if curvature > 40:  # ← 急弯曲率阈值
        curvature_factor = 0.5  # ← 急弯前瞻比例
    elif curvature > 20:  # ← 中等弯道曲率阈值
        curvature_factor = 0.7  # ← 中等弯道前瞻比例
    elif curvature > 10:  # ← 缓弯曲率阈值
        curvature_factor = 0.85  # ← 缓弯前瞻比例
    else:  # 直道
        curvature_factor = 1.0  # ← 直道前瞻比例
        
    adaptive_lookahead = base_lookahead * curvature_factor
    
    return max(self.min_lookahead_distance, 
              min(adaptive_lookahead, self.max_lookahead_distance))
